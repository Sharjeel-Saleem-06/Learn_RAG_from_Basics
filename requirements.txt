# ============================================
# RAG Chatbot - Dependencies
# ============================================

# --- Document Loading ---
# PyPDF2: Library to read and extract text from PDF files
# Think of it as a "PDF reader" for Python
PyPDF2>=3.0.0

# python-docx: Library to read Microsoft Word (.docx) files
python-docx>=0.8.11

# --- Text Processing ---
# tiktoken: OpenAI's tokenizer — counts how many "tokens" are in your text
# Why? Because LLMs have token LIMITS. You need to know how many tokens you're sending.
tiktoken>=0.5.0

# --- Embeddings ---
# sentence-transformers: FREE, LOCAL embedding models (no API key needed!)
# These convert your text into numerical vectors
sentence-transformers>=2.2.0

# openai: OpenAI's library for both embeddings AND chat completion
# You'll need an API key for this one
openai>=1.0.0

# --- Vector Database ---
# chromadb: An open-source vector database that runs LOCALLY
# Perfect for learning — no cloud setup needed
chromadb>=0.4.0

# --- LLM Integration ---
# groq: Groq's API client — they offer FREE LLM access (fast inference)
# Great for learning without spending money
groq>=0.4.0

# --- Web Framework ---
# fastapi: Modern, fast web framework for building APIs
# This is what your frontend will talk to
fastapi>=0.100.0

# uvicorn: ASGI server to run FastAPI applications
# Think of it as the "engine" that runs your web server
uvicorn>=0.23.0

# --- Frontend ---
# streamlit: Quick way to build interactive web UIs with Python
# Perfect for AI demos and prototypes
streamlit>=1.28.0

# --- Utilities ---
# python-dotenv: Load environment variables from a .env file
# This is where you'll store API keys securely
python-dotenv>=1.0.0

# python-multipart: Required by FastAPI for file uploads
python-multipart>=0.0.6

# rich: Beautiful terminal output for debugging
rich>=13.0.0
